йҰ–е…Ҳе®ҡд№үеӯҗеәҸеҲ—пјҢеҪўејҸеҢ–е®ҡд№үдёәпјҡз»ҷе®ҡдёҖдёӘеәҸеҲ—$X=\{x_1,x_2,вҖҰ,x_m\}$пјҢеҸҰдёҖдёӘеәҸеҲ—$Z=\{z_1,z_2,вҖҰ,z_k\}$ж»Ўи¶іеҰӮдёӢжқЎд»¶ж—¶з§°дёәXзҡ„еӯҗеәҸеҲ—пјҢеҚіеӯҳеңЁдёҖдёӘдёҘж јйҖ’еўһзҡ„Xзҡ„дёӢж ҮеәҸеҲ—$\{i_1,i_2,вҖҰ,i_k\}$пјҢеҜ№жүҖжңү$j=1,2,вҖҰ,k$пјҢж»Ўи¶іпјҢ$x_{i_j} = z_j$дҫӢеҰӮпјҢ$Z=\{B,C,D,B\}$жҳҜ$X=\{A,B,C,B,D,A,B\}$зҡ„еӯҗеәҸеҲ—пјҢеҜ№еә”зҡ„дёӢж ҮеәҸеҲ—дёә$\{2,3,5,7\}$гҖӮ

з»ҷе®ҡдёӨдёӘеәҸеҲ— X е’Ң YпјҢеҰӮжһң Z ж—ўжҳҜ X зҡ„еӯҗеәҸеҲ—пјҢд№ҹжҳҜ Y зҡ„еӯҗеәҸеҲ—пјҢеҲҷз§°е®ғжҳҜ X е’Ң Y зҡ„е…¬е…ұеӯҗеәҸеҲ—гҖӮ

жұӮдёӨдёӘеәҸеҲ—зҡ„жңҖй•ҝе…¬е…ұеӯҗеәҸеҲ—пјҢеҸҜд»ҘдҪҝз”ЁеҠЁжҖҒи§„еҲ’жұӮи§ЈгҖӮ

еҲ»з”»жңҖй•ҝе…¬е…ұеӯҗеәҸеҲ—зү№еҫҒ

пјҲLCS зҡ„жңҖдјҳеӯҗз»“жһ„пјүд»Ө$X=\{x_1,x_2,вҖҰ,x_m\}$е’Ң$Y=\{y_1,y_2,вҖҰ,y_n\}$дёәдёӨдёӘеәҸеҲ—пјҢ$Z=\{z_1,z_2,вҖҰ,z_k\}$дёә X е’Ң Y зҡ„д»»ж„Ҹ LCSгҖӮ

  1. еҰӮжһң$x_m = y_n$пјҢеҲҷ$z_k=x_m=y_n$пјҢдё”$Z_{k-1}$жҳҜ$X_{m-1}$е’Ң$Y_{n-1}$зҡ„дёҖдёӘ LCSпјӣ
  2. еҰӮжһң$x_m \neq y_n$пјҢйӮЈд№Ҳ$z_k \neq x_m$ж„Ҹе‘ізқҖ Z жҳҜ$X_{m-1}$е’Ң Y зҡ„дёҖдёӘ LCSпјӣ
  3. еҰӮжһң$x_m \neq y_n$пјҢйӮЈд№Ҳ$z_k \neq y_n$ж„Ҹе‘ізқҖ Z жҳҜ$Y_{n-1}$е’Ң Y зҡ„дёҖдёӘ LCSгҖӮ

дёҖдёӘйҖ’еҪ’и§Ј

е®ҡд№ү$c[i,j]$иЎЁзӨә$X_i$е’Ң$Y_j$зҡ„ LCS й•ҝеәҰпјҢж №жҚ®жңҖдјҳеӯҗз»“жһ„жҖ§иҙЁжҳ“еҫ—пјҡ
$\begin{equation}\label{Dirichlet}
c[i,j]=\begin{cases}
0, & \text{ иӢҘ } i=1жҲ–i=0\\
c[i-1,j-1] + 1, & \text{ иӢҘ } i,j >0дё”x_i = y_j\\
max(c[i-1,j], c[i,j-1]), & \text{ иӢҘ } i,j >0дё”x_i \neq y_j
\end{cases}\end{equation}$

и®Ўз®— LCS зҡ„й•ҝеәҰ

з”ұдёҠејҸеҸҜд»ҘеҫҲе®№жҳ“еҶҷеҮәдёҖдёӘжҢҮж•°ж—¶й—ҙзҡ„йҖ’еҪ’з®—жі•жқҘи®Ўз®—гҖӮдҪҶжҳҜ LCS й—®йўҳеҸӘжңү$\Theta (mn)$дёӘеӯҗй—®йўҳпјҢеҸҜд»ҘдҪҝз”ЁеҠЁжҖҒи§„еҲ’иҮӘеә•еҗ‘дёҠи®Ўз®—гҖӮиҝҗиЎҢж—¶й—ҙдёә$\Theta (mn)$гҖӮ

еҒҮи®ҫдёҖдёӘиЎЁж јCпјҢc[i,j]иЎЁзӨә$A_i$е’Ң$B_j$зҡ„жңҖй•ҝ LCSгҖӮ

C++е®һзҺ°д»Јз ҒеҰӮдёӢпјҡ

#include <iostream>
#include <vector>

using namespace std;

void get_lcs(string x, string y, vector<vector<int>> &c) {
    for (int i = 0; i < x.size(); ++i)
        for (int j = 0; j < y.size(); ++j) {
            if (x[i] == y[j])
                c[i + 1][j + 1] = c[i][j] + 1;
            else
                c[i + 1][j + 1] = c[i][j + 1] > c[i + 1][j] ? c[i][j + 1] : c[i + 1][j];
        }
}

void print_lcs(string x, vector<vector<int>> c) {
    int i = c.size() - 1;
    int j = c[0].size() -1;
    string result = "";
    while (i > 0 && j > 0) {
        if (c[i][j] > c[i - 1][j] && c[i][j] > c[i][j - 1]) {
            i--;
            j--;
            result = result + x[i];
        }
        else if (c[i][j] > c[i - 1][j]) {
            j--;
        }
        else
            i--;
    }
    cout << result;
}

int main() {
    string x = "10010101";
    string y = "010110110";
    vector<vector<int>> c;
    for (int i = 0; i <= x.size(); ++i) {
        vector<int> tmp;
        tmp.resize(y.size() + 1);
        c.push_back(tmp);
    }
    get_lcs(x, y, c);
    print_lcs(x, c);
    return 0;
}
